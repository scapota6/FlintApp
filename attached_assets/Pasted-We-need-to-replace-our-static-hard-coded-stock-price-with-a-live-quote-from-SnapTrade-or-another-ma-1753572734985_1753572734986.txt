     ```ts
     // GET /api/snaptrade/quote?symbol=AAPL
     router.get('/quote', isAuthenticated, async (req, res, next) => {
       try {
         const sym = (req.query.symbol as string).toUpperCase();
         // Use SnapTrade referenceData to fetch a real-time quote
         const { data } = await snapTradeClient.referenceData.getRealTimeQuote({
           symbol: sym
         });
         // Return { symbol: string; price: number; timestamp: string }
         res.json({ symbol: sym, price: data.lastPrice, timestamp: data.timestamp });
       } catch (err) {
         next(err);
       }
     });
     ```
